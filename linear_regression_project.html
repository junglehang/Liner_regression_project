def matxMultiply(A, B):
    row_a,clo_a = shape(A)
    row_b,clo_b = shape(B)
    if clo_a == row_b:
        source = []
        for i in range(row_a):
            source.append([])
            for j in range(clo_b):
                num_sum = 0
                for k in range(clo_a):
                    result = A[i][k] * B[k][j]
                    num_sum += result
                source[i].append(num_sum)
        return source
    else:
        raise ValueError

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        return None
    elif len(A) != len(A[0]):
        return None
    else:
        Ab = augmentMatrix(A,b)
        matxRound(Ab,decPts)
        row_ab,clo_ab = shape(Ab)
        for c in range(clo_ab-1):
            num_max = 0
            current_row = c
            for j in range(c,row_ab):
                if abs(Ab[j][c]) > num_max:
                    num_max = abs(Ab[j][c])
                    current_row = j
            if abs(num_max) < epsilon:
                return None
            else:
                swapRows(Ab,c,current_row)
                while abs((Ab[c][c]-1.0)) >= epsilon:
                    scaleRow(Ab,c,1.0 / Ab[c][c])
                for i in range(c):
                    while abs(Ab[i][c]) >= epsilon:
                        addScaledRow(Ab,i,c,-Ab[i][c])
                for i in range(c+1,row_ab):
                    while abs(Ab[i][c]) >= epsilon:
                        addScaledRow(Ab,i,c,-Ab[i][c])
        end_loc = []
        for i in range(row_ab):
            end_loc.append([Ab[i][-1]])
        return end_loc

def calculateMSE2D(X,Y,m,b):
    if len(X) == len(Y) and len(X)!=0:
        n = len(X)
        result = [(Y[i]-(m*X[i])-b)**2 for i in range(n)]
        return sum(result) / float(n)
    else:
